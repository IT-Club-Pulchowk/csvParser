177CSV_PARSER_DEFN_API CSV_PARSER_Bool csv_deserialize_length_string(void *context, uint8_t *value, size_t len, CSV_PARSER_STRING *out)
178{
179 if (len) {
180 if (value[0] != '"') {
181 out->data = value;
182 out->len = len;
183 return 1;
184 }
185
186 value[len - 1] = 0; // removing quote at the end
187 out->data = value + 1;
188 out->len = len - 1;
189 return 1;
190 }

259CSV_PARSER_DEFN_API CSV_PARSER_Bool csv_deserialize_boolean(void *context, uint8_t *value, size_t len, CSV_PARSER_Bool *out)
260{
261 switch (len) {
262 case 1:
263 *out = (value[0] == '1');
264 return value[0] == '0' || value[0] == '1';
265
266 case 4:
267 if ((value[0] == 't' || value[0] == 'T') &&
268 (value[1] == 'r' || value[1] == 'R') &&
269 (value[2] == 'u' || value[2] == 'U') &&
270 (value[3] == 'e' || value[3] == 'E')) {
271 *out = 1;
272 return 1;
273 }
274
275 case 5:
276 if ((value[0] == 'f' || value[0] == 'F') &&
277 (value[1] == 'a' || value[1] == 'A') &&
278 (value[2] == 'l' || value[2] == 'L') &&
279 (value[3] == 's' || value[3] == 'S') &&
280 (value[4] == 'e' || value[4] == 'E')) {
281 *out = 0;
282 return 1;
283 }
284 }
285
286 return 0;
287}

308CSV_PARSER_DEFN_API size_t csv_deserialize(void *context, void *ptr_to_struct, CSV_DESERIALIZE_DESC *desc, size_t stride, CSV_PARSER *parser, size_t no_of_records) {
309 CSV_PARSER_ASSERT(ptr_to_struct != NULL && desc != NULL);
310
311 size_t length = 0;
312 // Fill 1 row each a time
313
314 CSV_PARSER_ASSERT(desc->length == parser->columns);
315
316 size_t lines_to_parse = parser->lines - 1; // 1st row consumed by the header information
317
318 size_t records_to_parse = lines_to_parse;
319 if (no_of_records < records_to_parse)
320 {
321 records_to_parse = no_of_records;
322 }
323
324 for (size_t record = 0; record < records_to_parse; ++record)
325 {
326
327 for (size_t col = 0; col < parser->columns; ++col)
328 {
329 uint8_t *value = csv_parser_next(parser, &length);
330 desc->deserializer[col](context, value, length, (char *)ptr_to_struct + desc->offset[col]);
331 }
332 ptr_to_struct = (char *)ptr_to_struct + stride;
333 }
334
335 return records_to_parse;
336}
