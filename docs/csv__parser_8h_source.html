234static int64_t _csv_parser_count_columns(CSV_PARSER *parser) {
235 int64_t count = 0;
236
237 uint8_t *end = parser->buffer + parser->buffer_length;
238
239 uint8_t *start = parser->position;
240
241 while (parser->position < end && *parser->position != '\n') {
242 switch (*parser->position) {
243 case '"': // field inside double quotes
244 while (parser->position < end && *parser->position != '"') {
245 if (*parser->position == '\n' || *parser->position == '\r') {
246 parser->error.reason = "Could not find matching \".";
247 parser->error.column = parser->position - start;
248 return -1;
249 }
250 parser->position += 1;
251 }
252 if (parser->position == end) {
253 parser->error.reason = "Could not find matching \". Reached end of file.";
254 parser->error.column = parser->position - start;
255 return -1;
256 }
257 break;
258 case '\'': // field inside single quotes
259 while (parser->position < end && *parser->position != '\'') {
260 if (*parser->position == '\n' || *parser->position == '\r') {
261 parser->error.reason = "Could not find matching '.";
262 parser->error.column = parser->position - start;
263 return -1;
264 }
265 parser->position += 1;
266 }
267 if (parser->position == end) {
268 parser->error.reason = "Could not find matching '. Reached end of file.";
269 parser->error.column = parser->position - start;
270 return -1;
271 }
272 break;
273 case ',':
274 *parser->position = '\0';
275 start = parser->position + 1;
276 count++;
277 break;
278 default:
279 break;
280 }
281
282 parser->position += 1;
283 }
284
285 // If we have found some number of elements and '\n' at the end,
286 // the total number of columns needs to be increased by 1
287 count += (count != 0);
288
289 return count;
290}

292static CSV_PARSER_Bool _csv_parser_count_lines_and_columns(CSV_PARSER *parser) {
293 CSV_PARSER_ASSERT(parser->buffer && parser->position);
294
295 int64_t columns = _csv_parser_count_columns(parser);
296 if (columns == -1) {
297 parser->error.line = 1;
298 return 0;
299 }
300 else if (columns == 0) {
301 parser->error.line = 1;
302 parser->error.reason = "Bad CSV file";
303 }
304
305 uint8_t *end = parser->buffer + parser->buffer_length;
306
307 while (parser->position < end && _CSV_PARSER_ISSPACE(*parser->position)) {
308 *parser->position = '\0';
309 parser->position += 1;
310 }
311
312 int64_t next_columns;
313 while ((next_columns = _csv_parser_count_columns(parser))) {
314 if (next_columns == -1 || columns != next_columns) {
315 parser->error.line = parser->lines + 1;
316 if (columns != next_columns) {
317 parser->error.reason = "Not enough number of values.";
318 }
319 return 0;
320 }
321
322 parser->lines += 1;
323
324 while (parser->position < end && _CSV_PARSER_ISSPACE(*parser->position)) {
325 *parser->position = '\0';
326 parser->position += 1;
327 }
328 }
329
330 parser->lines += 1;
331 parser->columns = columns;
332 parser->position = parser->buffer;
333 return 1;
334}

387 CSV_PARSER_ASSERT(fp);
388 size_t buffer_length = _csv_parser_get_file_size(fp);
389 uint8_t *buffer = csv_parser_malloc((buffer_length + 1) * sizeof(*buffer), parser->allocator_context);
390 if (buffer == NULL) {
391 parser->error.reason = "Allocation failed. Out of memory.";
392 return 0;
393 }
394 buffer[buffer_length] = 0;
395 size_t result = fread(buffer, buffer_length, 1, fp);
396
397 if (result != 1) {
398 parser->error.reason = "File could not be read.";
399 return 0;
400 }
401 return csv_parser_load_buffer(parser, buffer, buffer_length);
402}

421CSV_PARSER_DEFN_API uint8_t *csv_parser_next(CSV_PARSER *parser, size_t *length) {
422 uint8_t *end = parser->buffer + parser->buffer_length;
423
424 while (parser->position < end && !*parser->position)
425 parser->position += 1;
426
427 uint8_t *next_token = parser->position;
428
429 while (parser->position < end && *parser->position)
430 parser->position += 1;
431
432 *length = parser->position - next_token;
433
434 return next_token;
435}
