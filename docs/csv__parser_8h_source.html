223static int64_t _csv_parser_count_columns(CSV_PARSER *parser) {
224 int64_t count = 0;
225
226 uint8_t *end = parser->buffer + parser->buffer_length;
227
228 uint8_t *start = parser->position;
229
230 while (parser->position < end && *parser->position != '\n') {
231 switch (*parser->position) {
232 case '"': // field inside double quotes
233 while (parser->position < end && *parser->position != '"') {
234 if (*parser->position == '\n' || *parser->position == '\r') {
235 parser->error.reason = "Could not find matching \".";
236 parser->error.column = parser->position - start;
237 return -1;
238 }
239 parser->position += 1;
240 }
241 if (parser->position == end) {
242 parser->error.reason = "Could not find matching \". Reached end of file.";
243 parser->error.column = parser->position - start;
244 return -1;
245 }
246 break;
247 case '\'': // field inside single quotes
248 while (parser->position < end && *parser->position != '\'') {
249 if (*parser->position == '\n' || *parser->position == '\r') {
250 parser->error.reason = "Could not find matching '.";
251 parser->error.column = parser->position - start;
252 return -1;
253 }
254 parser->position += 1;
255 }
256 if (parser->position == end) {
257 parser->error.reason = "Could not find matching '. Reached end of file.";
258 parser->error.column = parser->position - start;
259 return -1;
260 }
261 break;
262 case ',':
263 *parser->position = '\0';
264 start = parser->position + 1;
265 count++;
266 break;
267 default:
268 break;
269 }
270
271 parser->position += 1;
272 }
273
274 // If we have found some number of elements and '\n' at the end,
275 // the total number of columns needs to be increased by 1
276 count += (count != 0);
277
278 return count;
279}

281static CSV_PARSER_Bool _csv_parser_count_lines_and_columns(CSV_PARSER *parser) {
282 CSV_PARSER_ASSERT(parser->buffer && parser->position);
283
284 int64_t columns = _csv_parser_count_columns(parser);
285 if (columns == -1) {
286 parser->error.line = 1;
287 return 0;
288 }
289 else if (columns == 0) {
290 parser->error.line = 1;
291 parser->error.reason = "Bad CSV file";
292 }
293
294 uint8_t *end = parser->buffer + parser->buffer_length;
295
296 while (parser->position < end && _CSV_PARSER_ISSPACE(*parser->position)) {
297 *parser->position = '\0';
298 parser->position += 1;
299 }
300
301 int64_t next_columns;
302 while ((next_columns = _csv_parser_count_columns(parser))) {
303 if (next_columns == -1 || columns != next_columns) {
304 parser->error.line = parser->lines + 1;
305 if (columns != next_columns) {
306 parser->error.reason = "Not enough number of values.";
307 }
308 return 0;
309 }
310
311 parser->lines += 1;
312
313 while (parser->position < end && _CSV_PARSER_ISSPACE(*parser->position)) {
314 *parser->position = '\0';
315 parser->position += 1;
316 }
317 }
318
319 parser->lines += 1;
320 parser->columns = columns;
321 parser->position = parser->buffer;
322 return 1;
323}

375CSV_PARSER_DEFN_API CSV_PARSER_Bool csv_parser_load_file(CSV_PARSER *parser, FILE *fp) {
376 CSV_PARSER_ASSERT(fp);
377 size_t buffer_length = _csv_parser_get_file_size(fp);
378 uint8_t *buffer = csv_parser_malloc((buffer_length + 1) * sizeof(*buffer), parser->allocator_context);
379 if (buffer == NULL) {
380 parser->error.reason = "Allocation failed. Out of memory.";
381 return 0;
382 }
383 buffer[buffer_length] = 0;
384 size_t result = fread(buffer, buffer_length, 1, fp);
385
386 if (result != 1) {
387 parser->error.reason = "File could not be read.";
388 return 0;
389 }
390 return csv_parser_load_buffer(parser, buffer, buffer_length);
391}

410CSV_PARSER_DEFN_API uint8_t *csv_parser_next(CSV_PARSER *parser, size_t *length) {
411 uint8_t *end = parser->buffer + parser->buffer_length;
412
413 while (parser->position < end && !*parser->position)
414 parser->position += 1;
415
416 uint8_t *next_token = parser->position;
417
418 while (parser->position < end && *parser->position)
419 parser->position += 1;
420
421 *length = parser->position - next_token;
422
423 return next_token;
424}
